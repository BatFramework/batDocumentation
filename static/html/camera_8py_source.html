283 def update(self, dt):
284 if not self.follow_point_func or not (math.isfinite(dt) and dt > 0):
285 return

287 target = Vector2(self.follow_point_func())
288 self.vector_center.update(self.rect.center)

290 if self.damping == float("inf"):
